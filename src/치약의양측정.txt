#include <LiquidCrystal.h>

LiquidCrystal lcd(12, 11, 2, 3, 4, 5)

void setup(){
	pinMode(12, OUTPUT);
	pinMode(11, INPUT);
	lcd.begin(16, 2);
}

void loop(){
	digitalWrite(12, HIGH);
	delayMicroseconds(10);
	digitalWrite(12, LOW);

	long duration = pulseIn(11, HIGH);

	if(duration = 0){
		return ;
	}

	long distance = duration / 58.2;
	long value = (15 - distance) * (100 / 15)
	//치약의 크기가 15cm라고 가정

	lcd.clear();
	lcd.print(“toothpaste : ”);
	lcd.setCursor(0, 1);
	lcd.print(value);
	lcd.print(“%”);
	delay(500);
}
