#include "U8glib.h"

 //그래픽 LCD의 핀번호를 설정한다.
U8GLIB_NHD_C12864 u8g(13, 11, 10, 9, 8);    // SPI Com: SCK = 13, MOSI = 11, CS = 10, CD = 9, RST = 8

//초음파 센서의 핀번호를 설정한다.
int echoPin = 44;
int trigPin = 45;
int echoPin2 = 48;
int trigPin2 = 49;
int Bottle = 100;     //물통의 크기를 50cm라고 가정
int ToothPaste = 10;   //치약의 길이를 15cm라고 가정
 

//LCD 화면에 출력하게 하는 함수
void draw(long water, long tooth) {
  // graphic commands to redraw the complete screen should be placed here 
  u8g.setFont(u8g_font_unifont);
  //u8g.setFont(u8g_font_osb21);

  //물의 양 출력
  u8g.setPrintPos(0, 10);
  u8g.print("water:  ");
  u8g.setPrintPos(50, 10);
  u8g.print(water);
  u8g.setPrintPos(70, 10);
  u8g.print("%");

  //치약의 양 출력
  u8g.setPrintPos(0, 30);
  u8g.print("tooth:  ");
  u8g.setPrintPos(50, 30);
  u8g.print(tooth);
  u8g.setPrintPos(70, 30);
  u8g.print("%");
}

void clock(void){
  u8g.setPrintPos(50, 50);
  u8g.print("3 : 00");
  delay(1000);
  u8g.setPrintPos(50, 50);
  u8g.print("2 : 59");
}
 
void setup(void) {
   Serial.begin(9600);
  // trig를 출력모드로 설정, echo를 입력모드로 설정
  pinMode(trigPin, OUTPUT);
  pinMode(echoPin, INPUT);
  pinMode(trigPin2, OUTPUT);
  pinMode(echoPin2, INPUT);
  
  u8g.setContrast(0); // Config the contrast to the best effect
  u8g.setRot180();// rotate screen, if required
  // set SPI backup if required
  //u8g.setHardwareBackup(u8g_backup_avr_spi);
 
  // assign default color value
  if ( u8g.getMode() == U8G_MODE_R3G3B2 ) {
    u8g.setColorIndex(255);     // white
  }
  else if ( u8g.getMode() == U8G_MODE_GRAY2BIT ) {
    u8g.setColorIndex(3);         // max intensity
  }
  else if ( u8g.getMode() == U8G_MODE_BW ) {
    u8g.setColorIndex(1);         // pixel on
  }
  else if ( u8g.getMode() == U8G_MODE_HICOLOR ) {
    u8g.setHiColorByRGB(255,255,255);
  }
}
 
void loop(void) {
 float duration, distance;
 float duration2, distance2;
  
  // 초음파를 보낸다. 다 보내면 echo가 HIGH 상태로 대기하게 된다.
  digitalWrite(trigPin, HIGH);
  delay(1000);
  digitalWrite(trigPin, LOW);
  
  // echoPin 이 HIGH를 유지한 시간을 저장 한다.
  duration = pulseIn(echoPin, HIGH); 
  // HIGH 였을 때 시간(초음파가 보냈다가 다시 들어온 시간)을 가지고 거리를 계산 한다.
  distance = ((float)(340 * duration) / 10000) / 2;
  long water = (Bottle - distance) * (100 / Bottle);

  //시리얼 모니터의 남은 물의 양을 출력한다.
  Serial.println("*****water*****");
  Serial.print(water);
  Serial.println("%");
  Serial.print(distance);
  Serial.println("cm");
  delay(500);

  //초음파를 보낸다. 다 보내면 echo가 HIGH 상태로 대기하게 된다.
  digitalWrite(trigPin2, HIGH);
  delay(1000);
  digitalWrite(trigPin2, LOW);
  
  // echoPin 이 HIGH를 유지한 시간을 저장 한다.
  duration2 = pulseIn(echoPin2, HIGH); 
  // HIGH 였을 때 시간(초음파가 보냈다가 다시 들어온 시간)을 가지고 거리를 계산 한다.
  distance2 = ((float)(340 * duration2) / 10000) / 2;
  //치약의 길이가 10cm라고 가정
  long tooth = distance2 * (100 / ToothPaste);  

  Serial.println("*****tooth*****");
  Serial.print(tooth);
  Serial.println("%");
  Serial.print(distance2);
  Serial.println("cm");
  Serial.println(" ");
  
  // 수정한 값을 출력
  delay(500);
  
  // picture loop
  u8g.firstPage(); 
  do {
    draw(water, tooth);
    clock();
  }
  while( u8g.nextPage() );
 
  // rebuild the picture after some delay
  delay(500);
}
